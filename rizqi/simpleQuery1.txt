SELECT * FROM `tb_obat` WHERE jenis LIKE 'tablet'; 

SELECT * FROM `tb_obat` WHERE nama LIKE '%ml%'; 

SELECT * FROM `tb_obat` WHERE harga >= 10.000; 

SELECT * FROM `tb_obat` WHERE harga < 20.000 ORDER BY nama DESC; 

SELECT jenis, SUM(stok) as stok FROM `tb_obat` GROUP BY jenis; 

SELECT jenis, SUM(stok) as stok FROM `tb_obat` WHERE jenis LIKE '%cair%'; 

SELECT nama, jenis, harga FROM `tb_obat` WHERE jenis LIKE '%cair%' ORDER BY nama DESC; 

SELECT nama, jenis, harga FROM `tb_obat` WHERE harga > 100.000;
