-- 10 simpleQuery MySql --

--  SELECT --
1. SELECT * FROM barang;
 menampilkan seluruh tabel barang

2. SELECT nama_barang FROM barang;
 menampilkan kolom nama_barang di tabel barang

 -- WHERE --
3. SELECT * FROM barang WHERE harga_barang = 2000;
 menampilkan semua data di tabel barang dimana harganya sama dengan 2000

4. SELECT * FROM barang WHERE jenis_barang <> "botol";
 menampilkan semua data di tabel barang dimana jenis_barang tidak sama dengan botol (yang botol tidak di tampilkan)

5. SELECT * FROM barang WHERE harga_barang > 8000;
 menampilkan semua data di tabel barang yang harganya lebih besar dari 8000

6. SELECT * FROM barang WHERE stok_barang < 70;
 menampilkan semua data di tabel barang yang dimana stok_barang nya lebih kecil dari 70

7. SELECT * FROM barang WHERE harga_barang >= 8000;
 menampilkan semua data di tabel barang yang harganya lebih besar atau sama dengan 8000

8. SELECT * FROM barang WHERE harga_barang <= 8000;
 menampilkan semua data di tabel barang yang harganya lebih kecil atau sama dengan 8000

9. SELECT * FROM barang WHERE stok_barang BETWEEN 20 AND 70;
 menampilkan semua data di tabel barang yang dimana stok_barang nya diantara 20 hingga 70

10. SELECT * FROM barang WHERE nama_barang LIKE "pa%";
 menampilkan semua data di tabel barang yang dimana nama_barang seperti yg memiliki kata "pa" (panadol,paracetamol dll)
