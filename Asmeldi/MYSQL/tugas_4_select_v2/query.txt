// right Join and left Join

note 
JOIN 
    data yang ditampilkan obat.id_pembuat = pembuat.id_pembuat sama 2 punya hubungan/relasi

RIGHT JOIN 
    example = obat.id_pembuat = pembuat.id_pembuat
                   kiri              kanan

    jika lebih mengacu table sebelah kanan, jika table sebelah kanan punya relasi atau pun tidak
    punya relasi ke table sebelah kiri data -data  dari table sebelah kanan akan tetap ditampilkan semua
    

LEFT JOIN
    example = obat.id_pembuat = pembuat.id_pembuat
                   kiri              kanan

    jika lebih mengacu table sebelah kiri, jika table sebelah kiri punya relasi atau pun tidak
    punya relasi ke table sebelah kanan data - data  dari table sebelah kiri akan tetap ditampilkan semua
