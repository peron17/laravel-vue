8. INNER JOIN menggabungkan 2 tabel berelasi, kode yang atas sebagai tampilannya dan kode yang bawah sebagai kode relasinya
SELECT obat.id_obat, obat.harga_obat, transaksi.jumlah_transaksi
FROM obat
INNER JOIN transaksi ON obat.id_obat = transaksi.id_obat;

9. GROUPING + aritmatika
select id_pasien, id_obat, sum(jumlah_transaksi) as jml_transaksi
from
	transaksi
where
	id_pasien = '001'
group by id_pasien, id_obat;

10. ORDER BY mengurutkan data yang sesuai
select * from obat where stok_obat > 2000 order by stok_obat;
